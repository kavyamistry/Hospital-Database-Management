View: RoomAvgStay 
SELECT r.roomID, AVG((a.dischargeDate) - (a.admitDate)) AS AvgStay
                    FROM ROOM r
                    LEFT JOIN INPATIENT a ON r.roomID = a.roomID
                    GROUP BY a.inpatient_ID
